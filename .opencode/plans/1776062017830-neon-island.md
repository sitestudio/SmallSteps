describe("WordsAndSentences", () => {
  describe("Checkbox localStorage persistence", () => {
    it("should read checkbox state from localStorage", () => {
      // Setup
      const mockState = { lion: [true, false, true] };
      localStorage.getItem.and.returnValue(JSON.stringify(mockState));

      // Act
      const result = component.isItemChecked("item1");

      // Assert
      expect(result).toBe(true);
    });

    it("should save checkbox state to localStorage", () => {
      // Setup
      const mockEvent = { stopPropagation: jasmine.createSpy() } as any;
      const animalId = "lion";

      // Act
      component.handleCheck(mockEvent, "item1");

      // Assert
      const expectedState = {
        [animalId]: [true, false, false, false, false, false, false],
      };
      expect(localStorage.setItem).toHaveBeenCalledWith(
        "tinyStepsAnimalCheckboxes",
        JSON.stringify(expectedState),
      );
    });

    it("should initialize empty array for new animal", () => {
      // No existing state
      localStorage.getItem.and.returnValue(null);

      const mockEvent = { stopPropagation: jasmine.createSpy() } as any;
      component.handleCheck(mockEvent, "item3");

      // Should create array of 7 false values
      const expectedState = {
        lion: [false, false, true, false, false, false, false],
      };
      expect(localStorage.setItem).toHaveBeenCalledWith(
        "tinyStepsAnimalCheckboxes",
        JSON.stringify(expectedState),
      );
    });

    it("should clear all states on animal selection change", () => {
      // Setup - have some selected
      const mockEvent = { stopPropagation: jasmine.createSpy() } as any;
      component.handleCheck(mockEvent, "item1");

      // Reset animal selection
      localStorage.setItem(
        "tinyStepsSelectedAnimal",
        JSON.stringify({ selected: "tiger" }),
      );

      // Verify new animal has empty array
      expect(component.isItemChecked("item1")).toBe(false);
    });
  });

  describe("PDF generation", () => {
    it("should generate PDF with checked items per animal", () => {
      // Setup
      const mockState = {
        lion: [true, false, true],
        tiger: [false, true, false],
      };
      localStorage.getItem.and.returnValue(JSON.stringify(mockState));

      const doc = component.generatePDF();

      // Verify PDF contains expected content
      expect(doc.text).toHaveBeenCalledWith(
        jasmine.stringMatching(/ lion: 2 items checked/),
        jasmine.any(Number),
        jasmine.any(Number),
      );
    });

    it("should handle empty states gracefully", () => {
      localStorage.getItem.and.returnValue("{}");

      const doc = component.generatePDF();

      expect(doc.text).toHaveBeenCalledWith("No animals selected", 105, 60, {
        align: "center",
      });
    });
  });
});

test.describe("Words and Sentences Checkbox Persistence", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/");

    // Select an animal first
    const lionAnimal = page
      .locator(".animal-item")
      .filter({ hasText: "Lion" })
      .first();
    await lionAnimal.click();

    // Navigate to words-and-sentences
    await page.locator(".nav-item.item-1 a.nav-button").click();
    await page.getByText("Sounds and speech").click();
  });

  test("checkbox state persists in localStorage", async ({ page }) => {
    // Toggle a checkbox
    const firstCheckbox = page
      .locator('.checklist-item input[type="checkbox"]')
      .first();
    await firstCheckbox.click();

    // Verify checkbox is checked
    expect(await firstCheckbox.isChecked()).toBe(true);

    // Check localStorage directly
    const storageState = await page.evaluate(() => {
      return JSON.parse(
        localStorage.getItem("tinyStepsAnimalCheckboxes") || "{}",
      );
    });

    // Verify state was saved
    expect(Object.keys(storageState).length).toBeGreaterThan(0);
  });

  test("page reload maintains checkbox states", async ({ page }) => {
    // Toggle some checkboxes
    const checkboxes = page.locator('.checklist-item input[type="checkbox"]');
    await checkboxes.nth(0).click();
    await checkboxes.nth(2).click();
    await checkboxes.nth(4).click();

    // Reload page
    await page.reload();

    // Verify checkbox states persist
    expect(await checkboxes.nth(0).isChecked()).toBe(true);
    expect(await checkboxes.nth(2).isChecked()).toBe(true);
    expect(await checkboxes.nth(4).isChecked()).toBe(true);
  });

  test("generate PDF includes checked items", async ({ page }) => {
    // Check some items
    const checkboxes = page.locator('.checklist-item input[type="checkbox"]');
    await checkboxes.nth(1).click();
    await checkboxes.nth(3).click();

    // Generate PDF
    const [download] = await Promise.all([
      page.waitForEvent("download"),
      page.locator(".btn-primary").click(),
    ]);

    // Verify download created
    expect(download.suggestedFilename()).toBe(
      "words-and-sentences-checked-items.pdf",
    );
  });

  test("checkboxes persist across different animals", async ({ page }) => {
    const checkboxes = page.locator('.checklist-item input[type="checkbox"]');

    // Select first animal and check some items
    await checkboxes.nth(0).click();
    await checkboxes.nth(2).click();

    // Verify localStorage has first animal
    const storage1 = await page.evaluate(() => {
      return JSON.parse(
        localStorage.getItem("tinyStepsAnimalCheckboxes") || "{}",
      );
    });

    // Navigate to different animal (via home)
    await page.locator(".nav-btn.nav-home").click();

    // Select different animal
    const tigerAnimal = page
      .locator(".animal-item")
      .filter({ hasText: "Tiger" })
      .first();
    await tigerAnimal.click();

    // Navigate back
    await page.locator(".nav-btn.nav-back").click();

    // Verify different animal has empty array
    const storage2 = await page.evaluate(() => {
      return JSON.parse(
        localStorage.getItem("tinyStepsAnimalCheckboxes") || "{}",
      );
    });

    // Both animals should have entries
    expect(Object.keys(storage2).length).toBeGreaterThanOrEqual(2);
  });
});
